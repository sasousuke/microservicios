# Para exportar y cargar una imagen ya descargada y poder usarla en otra PC
https://aprendiendoavirtualizar.com/copiar-una-imagen-docker-a-otro-host/

# Exportar una imagen
docker save nombre_imagen > /directorio_destino/fichero.tar

# Comprimir la imagen con zip para disminuir tamaño del archivo
zip /directorio_destino/fichero.tar.zip /directorio_destino/fichero.tar

# Cargar una imagen
docker load -i /directorio/fichero.tar

# Descomprimir la imagen con unzip
unzip /directorio_destino/fichero.tar.zip

# Ejecutar docker compose con archivos separados de configuración

# Levantando el servicio declarado
docker compose -f mstream.yml up -d
docker compose -f jellyfin.yml up -d

# Deteniendo el servicio declarado
